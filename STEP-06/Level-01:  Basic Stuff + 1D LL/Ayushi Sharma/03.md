
### Deletion - At Begin, Middle and End

```cpp

#include<bits/stdc++.h>
using namespace std;

class Node {
    public:
        int data;
        Node* next;
        Node(int d) {
            data = d;
            next = NULL;
        }
};

// Q1 -- At Begin
Node* deleteFromStart(Node* head) {
    return head->next;
}


// Q2 -- At Given node
Node* deleteGivenNode(Node* head, int val) {
    
    if(head-> data == val) {
        return head->next;
    }

    Node* temp = head, *prev = NULL;
    while(temp->data != val) {
        prev = temp;
        temp = temp->next;
    }
    prev->next = temp->next;
    return head;
}


// Q3 -- At end
void deleteFromEnd(Node* head) {
    Node* temp = head;
    while(temp->next->next != NULL) {
        temp = temp->next;
    } 
    temp->next = NULL;
}


// Print (Traversal)
void printList(Node* head) {
    Node* temp = head;
    while(temp != NULL) {
        cout << temp->data << " ";
        temp = temp->next;
    }
    cout << "\n";
}


int main() {

    Node* head = new Node(10);
    Node* second = new Node(20);
    Node* third = new Node(30);
    Node* fourth = new Node(40);

    head->next = second;
    second->next = third;
    third->next = fourth;

    head = deleteFromStart(head);
    // head = deleteGivenNode(head,30);
    // deleteFromEnd(head);
    printList(head);

}
```
