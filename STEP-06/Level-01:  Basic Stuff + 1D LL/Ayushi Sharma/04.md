## Brute force approach: Find length and then traverse n/2 times

```cpp

class Solution {
public:
    ListNode* middleNode(ListNode* head) {
        
        // If the list is empty or has only one element, return the head as it's the middle.
        if (head == NULL || head->next == NULL) {
          return head;
        }

        ListNode* temp = head;
        int cnt = 0;

        while(temp != NULL) {
            cnt++;
            temp = temp -> next;
        }

        int half = cnt / 2;
        temp = head;

        while(half--) {
            temp = temp -> next;
        }
        return temp;
    }
};

// TC: O(N + N/2)
// SC: O(1)
```

## Optimal approach: 2 pointer (fast and slow)

```cpp
class Solution {
public:
    ListNode* middleNode(ListNode* head) {
        
        ListNode* slow = head;
        ListNode* fast = head;

        while(fast != NULL && fast->next != NULL) {
            slow = slow->next;
            fast = fast->next->next;
        }
        return slow;
    }
};

// TC: O(N/2)
// SC: O(1)
```
